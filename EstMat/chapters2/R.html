📈 Ley de Grandes Números

# Simular convergencia de la media
set.seed(123)
n <- 1:500
medias <- sapply(n, function(i) mean(rnorm(i, mean=10, sd=2)))
cat("Media final:", tail(medias, 1), "\n")
cat("Diferencia con μ=10:", abs(tail(medias, 1) - 10))

📐 Teorema Central del Límite

# Simular medias de exponenciales
set.seed(123)
medias <- replicate(1000, mean(rexp(30, rate=1)))
cat("Media de las medias:", mean(medias), "\n")
cat("Varianza de las medias:", var(medias), "\n")
cat("Varianza teórica (1/30):", 1/30)

📏 Intervalo de Confianza

# IC para la media con t.test

🔄 Bootstrap

# Estimar error estándar de la media
set.seed(123)
datos <- rnorm(100, mean=50, sd=10)
B <- 200
boot_means <- replicate(B, mean(sample(datos, replace=TRUE)))
cat("Error estándar bootstrap:", sd(boot_means), "\n")
cat("Error estándar teórico:", sd(datos)/sqrt(100))

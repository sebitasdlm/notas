📈 2. Ley de los Grandes Números (LGN)

Convergencia en probabilidad
Bernoulli
Normal
Cauchy (contraejemplo)
📖 Ley de los Grandes Números: La media muestral converge en probabilidad (y casi seguramente) a la media poblacional cuando n → ∞. 
# ====================================================
# 2.1 LGN PARA NORMAL(μ=10, σ=2)
# ====================================================

set.seed(123)
n_max <- 1000
medias <- numeric(n_max)

for(i in 1:n_max) {
  medias[i] <- mean(rnorm(i, mean = 10, sd = 2))
}

plot(1:n_max, medias, type = "l", col = "blue",
     xlab = "Tamaño de muestra (n)",
     ylab = "Media muestral",
     main = "LGN: Convergencia a μ = 10",
     lwd = 2)
abline(h = 10, col = "red", lwd = 3, lty = 2)

# ====================================================
# 2.2 LGN PARA BERNOULLI(PROCESO DE LA PROPORCIÓN)
# ====================================================

set.seed(123)
p <- 0.3
n_max <- 1000
proporciones <- numeric(n_max)

for(i in 1:n_max) {
  proporciones[i] <- mean(rbinom(i, 1, p))
}

plot(1:n_max, proporciones, type = "l", col = "blue",
     xlab = "n", ylab = "Proporción muestral",
     main = "LGN: Convergencia a p = 0.3",
     lwd = 2)
abline(h = p, col = "red", lwd = 3, lty = 2)

# ====================================================
# 2.3 LEY DÉBIL - PROBABILIDAD DE ERROR
# ====================================================

epsilon <- 0.05
n_values <- seq(50, 1000, by = 50)
prob_error <- sapply(n_values, function(n) {
  mean(abs(rbinom(1000, n, p)/n - p) > epsilon)
})

plot(n_values, prob_error, type = "b", 
     xlab = "n", ylab = expression(P(|hat(p)-p| > epsilon)),
     main = "Ley Débil: La probabilidad de error tiende a 0",
     col = "blue", pch = 16, lwd = 2)
abline(h = 0, col = "red", lty = 2)

# ====================================================
# 2.4 CONTRAEJEMPLO: DISTRIBUCIÓN CAUCHY
# ====================================================

# La Cauchy no tiene media, por lo que la LGN NO APLICA

set.seed(123)
cauchy_medias <- sapply(1:500, function(n) mean(rcauchy(n)))

plot(1:500, cauchy_medias, type = "l",
     xlab = "n", ylab = "Media muestral",
     main = "Cauchy: ¡No converge! (no tiene media)",
     col = "blue", lwd = 2)
abline(h = 0, col = "red", lty = 2)

📖 Teorema Central del Límite: La suma (o media) de variables i.i.d. con varianza finita converge en distribución a una normal.

🔢 4. Estimación Puntual

Máxima Verosimilitud (MLE)
Método de Momentos
Sesgo y varianza
ECM (Error Cuadrático Medio)

📖 Estimación puntual: Un estimador es una función de la muestra que aproxima un parámetro desconocido. 
# ====================================================
# 4.1 MLE PARA POISSON
# ====================================================

set.seed(123)
n <- 100
lambda_true <- 3
muestra <- rpois(n, lambda_true)

# MLE = media muestral
lambda_mle <- mean(muestra)
cat("MLE de λ:", lambda_mle, "\n")

# Función de log-verosimilitud
loglik_poisson <- function(lambda, x) {
  sum(dpois(x, lambda, log = TRUE))
}

# Optimización numérica
resultado <- optim(par = 1, fn = loglik_poisson, x = muestra,
                   control = list(fnscale = -1), method = "Brent",
                   lower = 0.1, upper = 10)
cat("Optimización numérica:", resultado$par, "\n")

# ====================================================
# 4.2 MLE PARA EXPONENCIAL
# ====================================================

muestra_exp <- rexp(n, rate = 2)
lambda_mle_exp <- 1 / mean(muestra_exp)
cat("MLE de λ (Exponencial):", lambda_mle_exp, "\n")

# ====================================================
# 4.3 MLE PARA NORMAL
# ====================================================

muestra_norm <- rnorm(n, mean = 5, sd = 2)
mu_mle <- mean(muestra_norm)
sigma_mle <- sqrt(mean((muestra_norm - mu_mle)^2))
cat("MLE de μ:", mu_mle, "\n")
cat("MLE de σ:", sigma_mle, "\n")

# ====================================================
# 4.4 SESGO, VARIANZA Y ECM
# ====================================================

# Simular sesgo del MLE
n_sim <- 1000
n_muestra <- 50
estimaciones <- replicate(n_sim, mean(rpois(n_muestra, lambda_true)))

sesgo <- mean(estimaciones) - lambda_true
var_est <- var(estimaciones)
ecm <- var_est + sesgo^2

cat("Sesgo:", sesgo, "\n")
cat("Varianza:", var_est, "\n")
cat("ECM:", ecm, "\n")

# ====================================================
# 4.5 INTERVALOS DE CONFIANZA POR BOOTSTRAP
# ====================================================

B <- 1000
lambda_boot <- replicate(B, {
  muestra_boot <- sample(muestra, replace = TRUE)
  mean(muestra_boot)
})

ic_bootstrap <- quantile(lambda_boot, c(0.025, 0.975))
cat("IC Bootstrap 95%:", ic_bootstrap, "\n")

hist(lambda_boot, breaks = 30, probability = TRUE,
     main = "Distribución Bootstrap de λ̂",
     xlab = expression(hat(lambda)), col = "lightblue")
abline(v = lambda_true, col = "red", lwd = 2)
abline(v = ic_bootstrap, col = "blue", lty = 2, lwd = 2)

📏 5. Intervalos de Confianza

# ====================================================
# 5.1 IC PARA LA MEDIA (σ CONOCIDA)
# ====================================================

set.seed(123)
n <- 30
sigma <- 2
muestra <- rnorm(n, mean = 10, sd = sigma)
xbar <- mean(muestra)
z <- qnorm(0.975)

LI <- xbar - z * sigma / sqrt(n)
LS <- xbar + z * sigma / sqrt(n)

cat("IC 95% para μ (σ conocida): (", round(LI, 4), ",", round(LS, 4), ")\n")

# ====================================================
# 5.2 IC PARA LA MEDIA (t-STUDENT)
# ====================================================

intervalo_t <- t.test(muestra, conf.level = 0.95)
intervalo_t$conf.int

# ====================================================
# 5.3 SIMULACIÓN DE COBERTURA DEL IC
# ====================================================

n_sim <- 1000
n_muestra <- 30
cobertura <- replicate(n_sim, {
  x <- rnorm(n_muestra, mean = 0, sd = 1)
  ic <- t.test(x)$conf.int
  ic[1] <= 0 & ic[2] >= 0
})
cat("Cobertura observada:", mean(cobertura), "\n")

# ====================================================
# 5.4 IC PARA LA VARIANZA (χ²)
# ====================================================

n <- length(muestra)
s2 <- var(muestra)
q1 <- qchisq(0.025, n-1)
q2 <- qchisq(0.975, n-1)

ic_var <- c((n-1)*s2/q2, (n-1)*s2/q1)
cat("IC 95% para σ²:", round(ic_var, 4), "\n")

# ====================================================
# 5.5 IC PARA PROPORCIÓN
# ====================================================

n_encuesta <- 500
exitos <- 230
prop.test(exitos, n_encuesta, conf.level = 0.95)

# IC sin continuidad (aproximación normal)
p_hat <- exitos / n_encuesta
se <- sqrt(p_hat * (1 - p_hat) / n_encuesta)
ic_normal <- p_hat + c(-1, 1) * qnorm(0.975) * se
cat("IC aproximado (normal):", round(ic_normal, 4), "\n")

# ====================================================

# ====================================================
# 5.10 IC BASADO EN LIKELIHOOD (PERFIL)
# ====================================================

# Para Poisson
loglik_poisson_perfil <- function(lambda, x) {
  sum(dpois(x, lambda, log = TRUE))
}

lambda_vals <- seq(2, 4, length.out = 100)
loglik_vals <- sapply(lambda_vals, loglik_poisson_perfil, x = muestra)
loglik_max <- max(loglik_vals)

# IC de perfil (χ²₁)
ic_likelihood <- lambda_vals[loglik_vals > loglik_max - qchisq(0.95, 1)/2]
cat("IC de perfil 95% para λ:", range(ic_likelihood), "\n")

Aproximación Normal a Binomial ▼ 
n <- 100
p <- 0.3
x <- 0:n
teorico <- dbinom(x, n, p)
aprox <- dnorm(x, mean = n*p, sd = sqrt(n*p*(1-p)))
plot(x, teorico, type = "h", col = "blue", lwd = 2,
     ylab = "Probabilidad", main = "Aproximación Normal a Binomial")
lines(x, aprox, col = "red", lwd = 2)
legend("topright", legend = c("Binomial", "Normal aprox"),
       col = c("blue", "red"), lwd = 2)

Simulación de cobertura del IC ▼ 
n_sim <- 1000
cobertura <- replicate(n_sim, {
  x <- rnorm(30, mean = 0, sd = 1)
  ic <- t.test(x)$conf.int
  ic[1] <= 0 & ic[2] >= 0
})
cat("Cobertura observada:", mean(cobertura))
